import express from "express";
import path from "path";
import { fileURLToPath } from "url";
import cors from "cors";

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const app = express();
app.use(cors());
app.use(express.json());

/** Health endpoints */
app.get("/healthz", (_req, res) => res.status(200).send("ok"));
app.get("/readyz", (_req, res) => res.status(200).send("ready"));

/** --- Bot endpoints --- */
app.post("/api/bot", async (req, res) => {
  const { botId, action, params } = req.body || {};
  try {
    // route to the right bot
    if (botId === "divine-oracle-bot" && action === "predict") {
      const out = await divineOraclePredict(params);
      return res.json({ ok: true, data: out });
    }
    if (botId === "guardian-angel-bot" && action === "analyze") {
      const out = await guardianAngelAnalyze(params);
      return res.json({ ok: true, data: out });
    }
    if (botId === "cyberlab-bot" && action === "scan") {
      const out = await cyberlabScan(params);
      return res.json({ ok: true, data: out });
    }
    return res.status(400).json({ ok: false, error: "Unknown bot/action" });
  } catch (e: any) {
    return res.status(500).json({ ok: false, error: e?.message || "Error" });
  }
});

/** --- Dashboard SPA (built files) --- */
const webDist = path.resolve(__dirname, "../../web/dist");
app.use("/dashboard", express.static(webDist));
app.get("/dashboard/*", (_req, res) =>
  res.sendFile(path.join(webDist, "index.html"))
);

/** Fallback for any other path */
app.get("/", (_req, res) => res.redirect("/dashboard"));

const port = Number(process.env.PORT || 3001);
app.listen(port, () => {
  console.log(`API listening on :${port}`);
});

8) Production smoke test (scripts/smoke.js)

This runs after deploy:

import http from "http";

function check(url) {
  return new Promise((resolve) => {
    http.get(url, (res) => resolve(res.statusCode)).on("error", () => resolve(0));
  });
}

const base = process.env.REPLIT_URL || "http://localhost:3001";
const tests = [
  `${base}/healthz`,
  `${base}/readyz`
];

const payload = JSON.stringify({
  botId: "divine-oracle-bot",
  action: "predict",
  params: { asset: "bitcoin", type: "crypto" }
});

function post(url) {
  return new Promise((resolve) => {
    const req = http.request(url, { method: "POST", headers: {"Content-Type":"application/json"} }, res => {
      resolve(res.statusCode);
    });
    req.on("error", () => resolve(0));
    req.write(payload);
    req.end();
  });
}

(async () => {
  const codes = await Promise.all(tests.map(check));
  const api = await post(`${base}/api/bot`);
  const ok = codes.every(c => c === 200) && api === 200;
  console.log(JSON.stringify({ health: codes, api, ok }, null, 2));
  process.exit(ok ? 0 : 1);
})();
